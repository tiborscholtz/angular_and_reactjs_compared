                        // .ts file
                        import React from 'react';

                        const ThemeSelector = () => {
                        let themeStyles;
                        let theme = "light";
                        switch (theme) {
                            case 'light':
                            themeStyles = {
                                backgroundColor: '#f4f4f4',
                                color: '#333',
                            };
                            break;
                            case 'dark':
                            themeStyles = {
                                backgroundColor: '#333',
                                color: '#fff',
                            };
                            break;
                            case 'blue':
                            themeStyles = {
                                backgroundColor: '#3498db',
                                color: '#fff',
                            };
                            break;
                            default:
                            themeStyles = {
                                backgroundColor: '#fff',
                                color: '#333',
                            };
                        }

                        return (
                            <div style=\{\{ padding: '20px', borderRadius: '8px', ...themeStyles }}>
                            <h2>Theme Selector</h2>
                            <p>This is a component with different themes.</p>
                            <p>Current Theme: {theme}</p>
                            </div>
                        );
                        };

                        // .ts file
                        import React, { useState, useEffect } from 'react';

                        const TodoList = () => {
                            const [todos, setTodos] = useState([]);
                            const [newTodo, setNewTodo] = useState('');

                            useEffect(() => {
                                    const fetchData = async () => {
                                    await new Promise(resolve => setTimeout(resolve, 1000));
                                    setTodos([
                                        { id: 1, text: 'Első' },
                                        { id: 2, text: 'Második' },
                                        { id: 3, text: 'Harmadik' },
                                    ]);
                            };
                                fetchData();
                            }, []);

                            const addTodo = () => {
                                if (newTodo.trim() !== '') {
                                    setTodos([...todos, { id: Date.now(), text: newTodo }]);
                                    setNewTodo('');
                                }
                            };

                            const removeTodo = (id) => {
                                setTodos(todos.filter(todo => todo.id !== id));
                            };

                            return (
                                <div>
                                    <h1>Todo List</h1>
                                    <ul>
                                        {todos.map(todo => (
                                            <li key={todo.id}>
                                            {todo.text}
                                            <button class="btn btn-primary" onClick={() => removeTodo(todo.id)}>Eltávolítás</button>
                                            </li>
                                        ))}
                                    </ul>
                                    <div>
                                        <input
                                        class="btn btn-primary"
                                        type="text"
                                        placeholder="Új todo"
                                        value={newTodo}
                                        onChange={(e) => setNewTodo(e.target.value)}
                                        />
                                        <button onClick={addTodo}>Hozzáadás</button>
                                    </div>
                                </div>
                            );
                        };

                        export default TodoList;
